    output = ""
    for i, point in enumerate(checklist, start=1):
        response = input(f"{i}. {point} (SI/NO/NON PERTINENTE): ")
        if response.upper() in ["SI", "NO", "NON PERTINENTE"]:
            output += f"{i}. {response}, "

            print("Invalid response. Please try again.")
            return None
    output += "\nNote finali: "

    return output
